function ProxyOverlay(str)
    growtopia.notify(str)

function openDialog(content)
    sendVariant({ v1 = "OnDialogRequest", v2 = content })

function wear(id)
    sendPacketRaw(false, { type = 10, value = id })
end

function cty(id, id2, amount)
    for _, inv in pairs(getInventory()) do
        if inv.id == id and inv.amount < amount then
            sendPacketRaw(false, { type = 10, value = id2 })
        end
    end
end

function posStr(px, py)
    if not px or not py then return "`4[Not Set]" end
    return "`b[X:"..px.." Y:"..py.."]"
end

function calcTax(amount, percent)
    if tonumber(percent) and tonumber(amount) then
        return math.floor((amount * percent) / 100)
    end
    return 0
end

function buildResult(i, b, d, wv)
    local r = ""
    if i and i ~= 0 then r = r..i.." `bBlack GL`0 " end
    if b and b ~= 0 then r = r..b.." `eBlue GL`0 " end
    if d and d ~= 0 then r = r..d.." `1DL`0 " end
    if wv and wv ~= 0 then r = r..wv.." `9WL`0 " end
    if r == "" then r = "`4Nothing" end
    return r
end

function calcAmount()
    Amount = 0
    for _, list in pairs(data) do
        if list.id == 11550 then
            Amount = Amount + list.count * 1000000
        elseif list.id == 7188 then
            Amount = Amount + list.count * 10000
        elseif list.id == 1796 then
            Amount = Amount + list.count * 100
        elseif list.id == 242 then
            Amount = Amount + list.count
        end
    end
    data = {}
end

function calcDrop(val)
    local v = tonumber(val) or 0
    local i = math.floor(v / 1000000)
    local tmp = v - i * 1000000
    local b = math.floor(tmp / 10000)
    tmp = tmp - b * 10000
    local d = math.floor(tmp / 100)
    local wv = tmp % 100
    return i, b, d, wv
end

function take()
    local tiles = {}
    if xt1 and yt1 then table.insert(tiles, {xt1, yt1}) end
    if xt2 and yt2 then table.insert(tiles, {xt2, yt2}) end
    if #tiles == 0 then
        ProxyOverlay("`4Set take position first! /st1 /st2")
        return
    end
    data = {}
    for _, obj in pairs(getObjectList()) do
        for _, t in pairs(tiles) do
            if obj.posX//32 == t[1] and obj.posY//32 == t[2] then
                sendPacketRaw(false, { type=11, value=obj.id, x=obj.posX, y=obj.posY })
                table.insert(data, { id=obj.itemid, count=obj.amount })
            end
        end
    end
    calcAmount()
    ProxyOverlay("`2Bet taken! `wTotal: `9"..Amount.." `wWL value")
end

function eattax(x, y)
    if not x or not y then return end
    local lx = getLocal().posX // 32
    local ly = getLocal().posY // 32
    if math.abs(lx - x) > 8 or math.abs(ly - y) > 8 then
        LogToConsole("`4[eattax] Too far from win position!")
        return
    end
    local t = getTile(x, y)
    if not t or t.collidable then return end
    local t1 = getTile(x+1, y)
    local t2 = getTile(x-1, y)
    local Z = 0
    if t1 and not t1.collidable then Z = 1
    elseif t2 and not t2.collidable then Z = -1
    else return end
    sendPacketRaw(false, { type=0, x=(x+Z)*32, y=y*32, state=(Z==1 and 48 or 32) })
end

function updateGemTile()
    gemTile = { pos1 = {}, pos2 = {} }
    if xgem1 then
        table.insert(gemTile.pos1, {x=xgem1, y=ygem1})
        if xgem2 then table.insert(gemTile.pos1, {x=xgem2, y=ygem2}) end
        if xgem3 then table.insert(gemTile.pos1, {x=xgem3, y=ygem3}) end
    end
    if xgemm1 then
        table.insert(gemTile.pos2, {x=xgemm1, y=ygemm1})
        if xgemm2 then table.insert(gemTile.pos2, {x=xgemm2, y=ygemm2}) end
        if xgemm3 then table.insert(gemTile.pos2, {x=xgemm3, y=ygemm3}) end
    end
end

function takegems()
    if #gemTile.pos1 == 0 and #gemTile.pos2 == 0 then
        ProxyOverlay("`4Set gem position first! /sh1 /sh2")
        return
    end
    local p1, p2 = 0, 0
    for _, obj in pairs(getObjectList()) do
        if obj.itemid == 112 then
            for _, t in pairs(gemTile.pos1) do
                if obj.posX//32 == t.x and obj.posY//32 == t.y then
                    p1 = p1 + obj.amount
                    sendPacketRaw(false, {type=11, value=obj.id, x=obj.posX, y=obj.posY})
                end
            end
            for _, t in pairs(gemTile.pos2) do
                if obj.posX//32 == t.x and obj.posY//32 == t.y then
                    p2 = p2 + obj.amount
                    sendPacketRaw(false, {type=11, value=obj.id, x=obj.posX, y=obj.posY})
                end
            end
        end
    end
    if p1 > p2 then
        sendPacket(2, "action|input\n|text|[(wow) `2WinP1`w] `2"..p1.." `b// `4"..p2.." `w[(lol) `4LoseP2`w]")
        ProxyOverlay("`2P1 WIN! `w"..p1.." vs `4"..p2)
    elseif p1 == p2 then
        sendPacket(2, "action|input\n|text|`0[(lol) TIE] `c"..p1.." `b// `c"..p2.." `0[(lol) TIE]")
        ProxyOverlay("`0TIE! `c"..p1.." vs "..p2)
    else
        sendPacket(2, "action|input\n|text|[(lol) `4LoseP1`w] `4"..p1.." `b// `2"..p2.." `w[(wow) `2WinP2`w]")
        ProxyOverlay("`2P2 WIN! `w"..p2.." vs `4"..p1)
    end
end

function qemefunc(number)
    local n = tonumber(number) or 0
    if n >= 10 then return tonumber(tostring(n):sub(-1)) end
    return n
end

function remefunc(number)
    local n = tonumber(number) or 0
    if n == 19 or n == 28 or n == 0 then return 0 end
    local s = tostring(math.floor(n/10) + n%10)
    return tonumber(s:sub(-1))
end

function getGame(num)
    if reme and not qeme then
        return "`b[`0REME:`9"..remefunc(num).."`b]"
    elseif not reme and qeme then
        return "`b[`0QEME:`9"..qemefunc(num).."`b]"
    end
    return ""
end

-- Drop 1 item dengan format GTPS yang benar
function dropItem(itemid, count)
    sendPacket(2, "action|dialog_return\ndialog_name|drop_item\nitemID|"..itemid.."|\ncount|"..count)
    sleep(300)
    sendPacket(2, "action|drop\n|itemID|"..itemid)
    sleep(600)
end

-- Drop semua item di thread baru
function doDrop(i, b, d, wv, onDone)
    runThread(function()
        if i and i > 0 then dropItem(11550, i) end
        if b and b > 0 then dropItem(7188, b) end
        if d and d > 0 then dropItem(1796, d) end
        if wv and wv > 0 then dropItem(242, wv) end
        ProxyOverlay("`2Drop DONE!")
        if onDone then onDone() end
    end)
end

-- Auto put chand setelah drop
-- Helper: taruh 1 chandelier di tile (tx, ty) dari posisi sekarang
-- Jalan ke dekat tile dulu agar tidak >5 block
function putOneTile(tx, ty)
    -- Jalan ke tile target dulu
    FindPath(tx, ty, 100)
    sleep(800) -- tunggu sampai benar-benar sampai
    -- Taruh chandelier tepat di tile tersebut
    local px = getLocal().posX
    local py = getLocal().posY
    sendPacketRaw(false, {
        type = 3,
        value = 5640,
        x = px,
        y = py,
        px = tx * 32,
        py = ty * 32,
        state = 16
    })
    sleep(600)
end

function doPutChand()
    if not putchand then return end
    if not (xgem1 and ygem1 and xgemm1 and ygemm1) then return end
    runThread(function()
        sleep(300)
        sendPacket(2, "action|input\ntext|`9[`bZlng`9] `9Putting Chand...")

        -- Taruh chandelier di posisi gem 1 (3 tile horizontal ke kiri)
        putOneTile(xgem1, ygem1)
        putOneTile(xgem1 - 1, ygem1)
        putOneTile(xgem1 - 2, ygem1)

        -- Taruh chandelier di posisi gem 2 (3 tile horizontal ke kanan)
        putOneTile(xgemm1, ygemm1)
        putOneTile(xgemm1 + 1, ygemm1)
        putOneTile(xgemm1 + 2, ygemm1)

        sendPacket(2, "action|input\n|text|`9[`bZlng`9] `2Put Chand DONE!")
        ProxyOverlay("`2Put Chand DONE!")
    end)
end

---=== SEND PACKET HOOK ===---
function Hook(ptype, str)
    if not str then return false end
    -- Debug log aktif
    if ptype == 2 then LogToConsole("DBG: "..str:sub(1,40)) end

    -- WRENCH BUTTON CLICK (intercept wrench popup dari server, bukan dari client)
    -- Packet dari server: "action|wrench" tidak ada, yang ada dari client adalah wrench packet
    -- Kita intercept di onSendPacket: client kirim wrench packet ke server
    if ptype == 2 and str:find("action|wrench") then
        local id = str:match("netid|(%d+)")
        if id then
            if pull then
                -- block packet asli, kirim pull
                sendPacket(2, "action|dialog_return\ndialog_name|popup\nnetID|"..id.."\nbuttonClicked|pull")
                LogToConsole("`9[`bZlng`9] `2Pulled netID: "..id)
                return true
            elseif kick then
                sendPacket(2, "action|dialog_return\ndialog_name|popup\nnetID|"..id.."\nbuttonClicked|kick")
                LogToConsole("`9[`bZlng`9] `4Kicked netID: "..id)
                return true
            elseif ban then
                sendPacket(2, "action|dialog_return\ndialog_name|popup\nnetID|"..id.."\nbuttonClicked|world_ban")
                LogToConsole("`9[`bZlng`9] `4Banned netID: "..id)
                return true
            end
        end
    end

    -- WRENCH PULL (/wrp atau button)
    if str:find("/wrp") or str:find("buttonClicked|wpull") then
        pull=true; kick=false; ban=false
        ProxyOverlay("`2PULL MODE ON")
        openDialog(getWRENCH())
        return true
    end

    -- WRENCH KICK (/wrk atau button)
    if str:find("/wrk") or str:find("buttonClicked|wkick") then
        kick=true; pull=false; ban=false
        ProxyOverlay("`4KICK MODE ON")
        openDialog(getWRENCH())
        return true
    end

    -- WRENCH BAN (/wrb atau button)
    if str:find("/wrb") or str:find("buttonClicked|wban") then
        ban=true; kick=false; pull=false
        ProxyOverlay("`4BAN MODE ON")
        openDialog(getWRENCH())
        return true
    end

    -- WRENCH OFF (/woff atau button) - hanya overlay, tidak buka menu lagi
    if str:find("/woff") or str:find("buttonClicked|woff") then
        ban=false; kick=false; pull=false
        ProxyOverlay("`4WRENCH MODE OFF")
        openDialog(getWRENCH())
        return true
    end

    -- CV DL KE BGL TOGGLE
    if str:find("/cv") or str:find("buttonClicked|cvdlkebgl") then
        cbgl = not cbgl
        openDialog(getMORE())
        if cbgl then
            sendPacket(2, "action|input\ntext|`9[`bZlng`9] `2Auto CV DL→BGL ON")

---=== VARIANT HOOK ===---
function onVar(var)
    if not var or not var.v1 then return false end

    -- World locked warning
    if var.v1 == "OnConsoleMessage" and var.v2 and var.v2:find("World Locked") then
        sendVariant({
            v1 = "OnAddNotification",
            v2 = "interface/atomic_button.rttex",
            v3 = "`w[`4WARNING`w] Always `2SAFE `wBefore `4PLAY",
            v4 = "audio/hub_open.wav"
        })
        sendPacket(2, "action|input\n|text|`w[`bPremium Proxy By : `9Zlng`w]")
    end

    -- Wheel spin console log
    if var.v1 == "OnConsoleMessage" and var.v2 and var.v2:find("spun the wheel") then
        ProxyLog("`w[`2REAL`w] "..var.v2)
        return true
    end

    -- Collected items announcement
    if var.v1 == "OnConsoleMessage" and var.v2 and var.v2:find("Collected") then
        local items = {
            {pattern="(%d+) World Lock",    name="World Lock", color="`9"},
            {pattern="(%d+) Diamond Lock",  name="Diamond Lock", color="`1"},
            {pattern="(%d+) Blue Gem Lock", name="Blue GL", color="`e"},
            {pattern="(%d+) Black Gem Lock",name="Black GL", color="`b"},
        }
        for _, item in pairs(items) do
            local j = var.v2:match(item.pattern)
            if j then
                LogToConsole("`9[`bZlng`9] `oCollected "..item.color..j.." "..item.name)
            end
        end
    end

    -- Respawn on kill
    if var.v1 == "OnKilled" or var.v1 == "OnKill" then
        sendPacket(2, "action|respawn")
        LogToConsole("`9[`bZlng`9] `6Auto Respawn!")
    end

    -- DL to BGL auto convert (hanya jika cbgl ON)
    if cbgl and var.v1 == "OnConsoleMessage" and var.v2 and var.v2:find("Diamond Lock") then
        local s = tonumber(var.v2:match("(%d+) Diamond Lock")) or 0
        if s >= 100 then
            for _, t in pairs(getTiles()) do
                if t.fg == 3898 then
                    sendPacket(2, "action|dialog_return\ndialog_name|telephone\nnum|53785|\nx|"..t.x.."|\ny|"..t.y.."|\nbuttonClicked|bglconvert")
                    ProxyOverlay("`2Auto CV: 100 DL → 1 BGL!")
                    break
                end
            end
        end
    end

    -- Block fast delivery dialog
    if var.v1 == "OnDialogRequest" and var.v2 and var.v2:find("Wow, that's fast delivery.") then
        return true
    end

    -- Console log semua pesan
    if var.v1 == "OnConsoleMessage" and var.v2 then
        ProxyLog("`b"..var.v2)
        return true
    end

    -- Block SDB
    if var.v1 == "OnSDBroadcast" and sdbb then
        sendPacket(2, "action|input\n|text|`9[`bZlng`9] `b// `8SKIP SDB `b//")
        ProxyOverlay("`b// `8SKIP SDB `b//")
        return true
    end

    -- Auto convert DL→BGL via telephone dialog
    if var.v1 == "OnDialogRequest" and var.v2 and var.v2:find("`wTelephone") and cbgl then
        local x = var.v2:match("embed_data|x|(%d+)")
        local y = var.v2:match("embed_data|y|(%d+)")
        if x and y then
            sendPacket(2, "action|dialog_return\ndialog_name|telephone\nnum|53785|\nx|"..x.."|\ny|"..y.."|\nbuttonClicked|bglconvert")
            sendPacket(2, "action|input\n|text|`9[`bZlng`9] `9Convert `1100 DL `9→ `e1 BGL")
        end
        return true
    end

    -- Fake wheel spin tag
    if var.v1 == "OnTalkBubble" and var.v3 and var.v3:find("spun the wheel") and var.v3:find("<") and var.v3:find(">") then
        sendVariant({v1=var.v1, v2=var.v2, v3=var.v3.." `w[`4FAKE`w]", v4=var.v4})
        return true
    end

    -- REME/QEME on talk bubble
    if var.v1 == "OnTalkBubble" and var.v3 and var.v3:find("spun the wheel") then
        if reme or qeme then
            local raw = var.v3:match("and got (.+)")
            if raw then
                local cleaned = raw:gsub("`",""):gsub("!%]",""):gsub("!7",""):gsub("]",""):gsub(" ","")
                local num = tonumber(cleaned:sub(2)) or tonumber(cleaned) or 0
                sendVariant({v1=var.v1, v2=var.v2, v3="`w[`2REAL`w] "..var.v3.." "..getGame(num), v4=var.v4})
                return true
            end
        end
        sendVariant({v1=var.v1, v2=var.v2, v3="`w[`2REAL`w] "..var.v3, v4=var.v4})
        return true
    end

    return false
end
