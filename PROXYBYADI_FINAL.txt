function eattax(x, y)
    if not x or not y then return end
    local lx = getLocal().posX // 32
    local ly = getLocal().posY // 32
    if math.abs(lx - x) > 8 or math.abs(ly - y) > 8 then return end
    local t = getTile(x, y)
    if not t or t.collidable then return end
    local t1 = getTile(x+1, y)
    local t2 = getTile(x-1, y)
    local Z = 0
    if t1 and not t1.collidable then Z = 1
    elseif t2 and not t2.collidable then Z = -1
    else return end
    sendPacketRaw(false, { type=0, x=(x+Z)*32, y=y*32, state=(Z==1 and 48 or 32) })
end

function qemefunc(number)
    local n = tonumber(number) or 0
    if n >= 10 then return tonumber(tostring(n):sub(-1)) end
    return n
end

function remefunc(number)
    local n = tonumber(number) or 0
    if n == 19 or n == 28 or n == 0 then return 0 end
    local s = tostring(math.floor(n/10) + n%10)
    return tonumber(s:sub(-1))
end

function getGame(num)
    if reme and not qeme then
        return "`b[`0REME: `9"..remefunc(num).."`b]"
    elseif not reme and qeme then
        return "`b[`0QEME: `9"..qemefunc(num).."`b]"
    end
    return ""
end

function calcDrop(val)
    local v = tonumber(val) or 0
    local i = math.floor(v / 1000000)
    local tmp = v - i*1000000
    local b = math.floor(tmp / 10000)
    tmp = tmp - b*10000
    local d = math.floor(tmp / 100)
    local wv = tmp % 100
    return i, b, d, wv
end

-- onVariant hanya 1 param (var) sesuai docs PowerKuy
function onVar(var)
    if not var or not var.v1 then return false end

    -- World locked warning
    if var.v1 == "OnConsoleMessage" and var.v2 and var.v2:find("World Locked") then
        sendVariant({
            v1 = "OnAddNotification",
            v2 = "interface/atomic_button.rttex",
            v3 = "`w[`4WARNING`w] Always `2SAFE `wBefore `4PLAY",
            v4 = "audio/hub_open.wav"
        })
        sendPacket(2, "action|input\n|text|`w[`bPremium Proxy By : `9Adi`w]")
    end

    -- Wheel spin log
    if var.v1 == "OnConsoleMessage" and var.v2 and var.v2:find("spun the wheel") then
        ProxyLog("`w[`2REAL`w] "..var.v2)
        return true
    end

    -- Collected World Lock - auto wear
    if var.v1 == "OnConsoleMessage" and var.v2 then
        if var.v2:find("Collected") then
            if var.v2:find("World Lock") then
                local j = var.v2:match("(%d+) World Lock")
                wear(242)
                if j then sendPacket(2, "action|input\n|text|`9[`bAdi`9] `oCollected `w"..j.." World Lock") end
            end
            if var.v2:find("Diamond Lock") then
                local j = var.v2:match("(%d+) Diamond Lock")
                if j then sendPacket(2, "action|input\n|text|`9[`bAdi`9] `oCollected `w"..j.." Diamond Lock") end
            end
            if var.v2:find("Blue Gem Lock") then
                local j = var.v2:match("(%d+) Blue Gem Lock")
                if j then sendPacket(2, "action|input\n|text|`9[`bAdi`9] `oCollected `w"..j.." Blue Gem Lock") end
            end
            if var.v2:find("Black Gem Lock") then
                local j = var.v2:match("(%d+) Black Gem Lock")
                if j then sendPacket(2, "action|input\n|text|`9[`bAdi`9] `oCollected `w"..j.." Black Gem Lock") end
            end
        end
    end

    -- Respawn on kill
    if var.v1 == "OnKill" then
        sendPacket(2, "action|input\n|text|`9[`bAdi`9] `6Respawn")
    end

    -- DL to BGL auto convert
    if var.v1 == "OnConsoleMessage" and var.v2 and var.v2:find("(%d+) Diamond Lock") then
        local jumlah = var.v2:match("(%d+) Diamond Lock")
        local s = tonumber(jumlah) or 0
        for _, t in pairs(getTiles()) do
            if t.fg == 3898 then
                for _, inv in pairs(getInventory()) do
                    if inv.id == 1796 and (inv.amount > 100 or s > 99) then
                        sendPacket(2, "action|dialog_return\ndialog_name|telephone\nnum|53785|\nx|"..t.x.."|\ny|"..t.y.."|\nbuttonClicked|bglconvert")
                        ProxyOverlay("`2Successfully Changed to Blue Gem Lock")
                    end
                end
            end
        end
    end

    -- Block fast delivery dialog
    if var.v1 == "OnDialogRequest" and var.v2 and var.v2:find("Wow, that's fast delivery.") then
        return true
    end

    -- Console message log
    if var.v1 == "OnConsoleMessage" and var.v2 then
        ProxyLog("`b"..var.v2)
        return true
    end

    -- Block SDB
    if var.v1 == "OnSDBroadcast" and sdbb then
        sendPacket(2, "action|input\n|text|`9[`bAdi`9] `b// `8SKIPED SDB `b//")
        ProxyOverlay("`b// `8SKIPED SDB `b//")
        return true
    end

    -- Auto convert DL to BGL on telephone dialog
    if var.v1 == "OnDialogRequest" and var.v2 and var.v2:find("`wTelephone") and cbgl then
        local x = var.v2:match("embed_data|x|(%d+)")
        local y = var.v2:match("embed_data|y|(%d+)")
        if x and y then
            sendPacket(2, "action|dialog_return\ndialog_name|telephone\nnum|53785|\nx|"..x.."|\ny|"..y.."|\nbuttonClicked|bglconvert")
            sendPacket(2, "action|input\n|text|`9[`bAdi`9] `9Convert `1100 Diamond Lock `9to `e1 Blue Gem Lock")
        end
        return true
    end

    -- Fake wheel spin tag
    if var.v1 == "OnTalkBubble" and var.v3 and var.v3:find("spun the wheel") and var.v3:find("<") and var.v3:find(">") then
        sendVariant({v1=var.v1, v2=var.v2, v3=var.v3.." `w[`4FAKE`w]", v4=var.v4})
        return true
    end

    -- REME/QEME on talk bubble
    if var.v1 == "OnTalkBubble" and var.v3 and var.v3:find("spun the wheel") then
        if reme or qeme then
            local raw = var.v3:match("and got (.+)")
            if raw then
                local cleaned = raw:gsub("`",""):gsub("!%]",""):gsub("!7",""):gsub("]",""):gsub(" ","")
                local num = tonumber(cleaned:sub(2)) or tonumber(cleaned) or 0
                sendVariant({v1=var.v1, v2=var.v2, v3="`w[`2REAL`w] "..var.v3.." "..getGame(num), v4=var.v4})
                return true
            end
        end
        sendVariant({v1=var.v1, v2=var.v2, v3="`w[`2REAL`w] "..var.v3, v4=var.v4})
        return true
    end

    return false
end
